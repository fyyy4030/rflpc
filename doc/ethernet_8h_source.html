276 static inline int rflpc_eth_get_current_tx_packet_descriptor(rflpc_eth_descriptor_t **descriptor, rflpc_eth_tx_status_t **status, int idx)
277 {
278  /* queue full */
279  if (TX_PRODUCE_INDEX_INC(idx) == LPC_EMAC->TxConsumeIndex - 1 ||
280  ((TX_PRODUCE_INDEX_INC(idx) == LPC_EMAC->TxDescriptorNumber) && LPC_EMAC->TxConsumeIndex == 0))
281  {
282  return 0;
283  }
284  *descriptor = ((rflpc_eth_descriptor_t*)LPC_EMAC->TxDescriptor) + TX_PRODUCE_INDEX_INC(idx);
285  *status = ((rflpc_eth_tx_status_t*)LPC_EMAC->TxStatus) + TX_PRODUCE_INDEX_INC(idx);
286  return 1;
287 }

400 static inline void rflpc_eth_activate_rx_filter(int accept_unicast, int accept_multicast, int accept_broadcast)
401 {
402  LPC_EMAC->RxFilterCtrl = 0;
403  if (accept_unicast)
404  LPC_EMAC->RxFilterCtrl |= RFLPC_ETH_RXFILTER_UNICAST_EN;
405  if (accept_multicast)
406  LPC_EMAC->RxFilterCtrl |= RFLPC_ETH_RXFILTER_MULTICAST_EN;
407  if (accept_broadcast)
408  LPC_EMAC->RxFilterCtrl |= RFLPC_ETH_RXFILTER_BROADCAST_EN;
409  LPC_EMAC->RxFilterCtrl |= RFLPC_ETH_RXFILTER_PERFECT_EN;
410  /* activate filter */
411  LPC_EMAC->Command &= ~RFLPC_ETH_CMD_PASS_RX_FILTER;
412 }
