89 static inline void rflpc_led_binary_value(uint8_t v)
90 {
91  uint32_t leds = 0;
92  if (v & 1)
93  leds |= RFLPC_LED_4;
94  if (v & 2)
95  leds |= RFLPC_LED_3;
96  if (v & 4)
97  leds |= RFLPC_LED_2;
98  if (v & 8)
99  leds |= RFLPC_LED_1;
100  rflpc_led_val(leds);
101 }
